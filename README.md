# WriteNow
Dating site, an analogue of tinder, but for friends

# Ruby version
3.0.2
# Rails version
7.0.4

# How to run the site
install yarn, nodejs, postgre<br />
git clone git@github.com:MupLever/Write_Now.git<br />
gem install bundle<br />
bundle<br />
yarn add bootstrap<br />
yarn upgrade

# Database creation
rails db:create<br />
rails db:migrate

# Start test 
bundle exec rspec
